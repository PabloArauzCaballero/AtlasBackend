function key(): Buffer {
  const raw = env.NOTIFICATION_TOKEN_ENCRYPTION_KEY ?? env.JWT_ACCESS_TOKEN_SECRET;
  return createHash('sha256').update(raw).digest();
}
 
export function encryptSecret(plainText: string): string {
